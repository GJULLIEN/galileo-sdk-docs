503 // Read the current configuration of a pin from the I/O Expander.
504 // This function assumes the caller has verified the pin number for range
505 // and has initialized the PIN_DATA struct passed in to zeroes.
506 //
507 // The "inline" marks the function as "selectany", while "__declspec (noinline)"
508 // keeps it from actually being expanded inline. Without the "inline" the linker
509 // will complain about multiple definitions of the same routine name.
510 __declspec (noinline) inline void _InitPinConfiguration(int pin, PPIN_DATA pinData)
511 {
512  int i;
513  BYTE portConfig[PORT_CONFIG_REG_COUNT] = { 0 };
514  BYTE pwmConfig[PWM_CONFIG_REG_COUNT] = { 0 };
515  BYTE retVal = 0;
516 
517  // Indicate which port we want the configuration of.
518  Wire.beginTransmission(CY8_ADDRESS);
519  Wire.write(CY8_PORT_SELECT);
520  Wire.write(_ArduinoToPortBitMap[pin].PortNumber);
521  Wire.endTransmission(TRUE); // Send STOP
522 
523  // Send the base address of the configuration registers.
524  Wire.beginTransmission(CY8_ADDRESS);
525  Wire.write(CY8_PORT_CONFIG_BASE);
526  Wire.endTransmission(FALSE); // Dont' send STOP
527 
528  // Read the configuration registers.
529  retVal = static_cast<BYTE>(Wire.requestFrom(CY8_ADDRESS, PORT_CONFIG_REG_COUNT));
530  if (retVal != PORT_CONFIG_REG_COUNT)
531  {
532  ThrowError("Error reported by Wire.requestFrom() for pin (%d).", pin);
533  }
534  for (i = 0; i < PORT_CONFIG_REG_COUNT; i++)
535  {
536  portConfig[i] = static_cast<BYTE>(Wire.read());
537  }
538 
539  // Determine whether the port pin is configured as an input or output.
540  // The I/O Expander uses 0 for output, 1 for input.
541  if ((portConfig[PIN_DIRECTION] & (1 << _ArduinoToPortBitMap[pin].BitNumber)) == 0)
542  {
543  pinData->currentMode = OUTPUT;
544  }
545  else
546  {
547  pinData->currentMode = INPUT;
548  }
549 
550  pinData->modeSet = INPUT; // This variable is used to track last explicitly set mode state, during initialization it is expected to be the default value - INPUT.
551 
552  // If the port pin is configured as a PMW:
553  if ((portConfig[SELECT_PWM] & (1 << _ArduinoToPortBitMap[pin].BitNumber)) != 0)
554  {
555  // If the pin is one on which we support PWM use:
556  if (_PwmPinMap[pin] != 0)
557  {
558  pinData->pwmIsEnabled = TRUE;
559 
560  // Indicate which PWM we want the configuration for.
561  Wire.beginTransmission(CY8_ADDRESS);
562  Wire.write(CY8_PWM_SELECT);
563  Wire.write(_ArduinoToPortBitMap[pin].PwmNumber);
564  Wire.endTransmission(TRUE); // Send STOP
565 
566  // Send the base address of the PWM configuration registers.
567  Wire.beginTransmission(CY8_ADDRESS);
568  Wire.write(CY8_PWM_CONFIG_BASE);
569  Wire.endTransmission(FALSE); // Don't send STOP
570 
571  // Read the PWM configuration registers.
572  retVal = static_cast<BYTE>(Wire.requestFrom(CY8_ADDRESS, PWM_CONFIG_REG_COUNT));
573  if (retVal != PWM_CONFIG_REG_COUNT)
574  {
575  ThrowError("Error reported by Wire.requestFrom() for pin (%d).", pin);
576  }
577  for (i = 0; i < PWM_CONFIG_REG_COUNT; i++)
578  {
579  pwmConfig[i] = static_cast<BYTE>(Wire.read());
580  }
581 
582  pinData->pwmDutyCycle = pwmConfig[PULSE_WIDTH_PWM];
583  }
584  // If we don't support PWM use on this pin, the user is on his own.
585  }
586 }

588 __declspec (noinline) inline void _InitPinMuxConfig(int pin, PPIN_DATA pinData)
589 {
590  unsigned int gpioMux;
591  BYTE muxConfig;
592  BYTE retVal = 0;
593 
594  // If this pin uses a MUX:
595  if (_ArduinoToPortBitMap[pin].MuxPort != NO_PORT)
596  {
597  // Read the output register that controlls the MUX.
598  Wire.beginTransmission(CY8_ADDRESS);
599  Wire.write(OUTPUT_PORTS_BASE + _ArduinoToPortBitMap[pin].MuxPort);
600  Wire.endTransmission(FALSE); // Don't send STOP
601 
602  // Read the output port register.
603  retVal = static_cast<BYTE>(Wire.requestFrom(CY8_ADDRESS, 1));
604  if (retVal != 1)
605  {
606  ThrowError("Error reported by Wire.requestFrom() for pin (%d).", pin);
607  }
608  muxConfig = static_cast<BYTE>(Wire.read());
609 
610  // Determine whether the MUX is set to default or alternate value.
611  muxConfig = (muxConfig >> _ArduinoToPortBitMap[pin].MuxBit) & 0x01;
612  gpioMux = _GalileoMuxMap[pin];
613  gpioMux = HIWORD(gpioMux);
614  if (muxConfig == gpioMux)
615  {
616  pinData->currentMux = DEFAULT_MUX;
617  }
618  else
619  {
620  pinData->currentMux = ALTERNATE_MUX;
621  }
622  }
623  else
624  {
625  pinData->currentMux = DEFAULT_MUX;
626  }
627  pinData->muxSet = DEFAULT_MUX; // This variable is used to track last explicitly set mux state, during initialization it is expected to be the default value - DEFAULT_MUX.
628 }

640 __declspec (noinline) inline void _InitializePin(int pin)
641 {
642  // Get the current state of the I2C MUX.
643  bool I2cWasEnabled = Wire.getI2cHasBeenEnabled();
644  bool done = false;
645 
646  Wire.begin();
647 
648  // Restore the state of the I2C MUX if it was changed by the Wire.begin() call.
649  if (!I2cWasEnabled)
650  {
651  GpioWrite(I2C_MUX, I2C_MUX_DISABLE);
652  Wire.setI2cHasBeenEnabled(false);
653  }
654 
655  try
656  {
657  // If the pin normally comes from the I/O Expander:
658  if (_ArduinoToGalileoPinMap[pin] >= GPORT0_BIT0_PWM7)
659  {
660  // Read the pin configuration from the CY8C9540A I/O Expander.
661  _InitPinConfiguration(pin, &(_pinData[pin]));
662 
663  // Indicate the data for this pin is now initialized.
664  _pinData[pin].pinInitialized = TRUE;
665 
666  done = true;
667  }
668 
669  // Read the MUX configuration for this pin.
670  _InitPinMuxConfig(pin, &(_pinData[pin]));
671  }
672  catch (const _arduino_fatal_error &)
673  {
674  done = false;
675  }
676 
677  // If the pin normally comes directly from the Quark SOC,
678  // or if an error occurred querying the pin configuration:
679  if (!done)
680  {
681  // We can't query the pin configuration, so we just initialize it.
682  _pinData[pin].pinInitialized = TRUE; // Avoid recursive loop calling pinMode()!

867 // Select a particular function on a multiplexed IO pin in such a way that it
868 // can be reverted later if needed.
869 // This function assumes the pin value has already been checked for range.
870 inline bool _SetImplicitPinFunction(ULONG pin, ULONG function)
871 {
872  _InitializePinIfNeeded(pin);
873 
874  unsigned int mux = _GalileoMuxMap[pin];
875 
876  if (mux != NOT_MUXED)
877  {
878  // pin supports alternate configuration. Select default
879  // or alternate configuration
880  int val = HIWORD(mux);
881  switch (function)
882  {
883  case DEFAULT_MUX:
884  // default function
885  break;
886  case ALTERNATE_MUX:
887  // alternate function
888  val = !val;
889  break;
890  default:
891  // unsupported function
892  return false;
893  }
894 
895  if (_pinData[pin].currentMux != function)
896  {
897  _ValidatePinOkToChange(pin);
898 
899  HRESULT hr = GpioSetDir(LOWORD(mux), 1);
900  if (FAILED(hr))
901  {
902  ThrowError("GpioSetDir() : Unexpected error");
903  }
904 
905  hr = GpioWrite(LOWORD(mux), val);
906  if (FAILED(hr))
907  {
908  ThrowError("GpioWrite() : Unexpected error");
909  }
910  _pinData[pin].currentMux = function;
911  }
912  return true;
913  }
914  else
915  {
916  // pin does not support alternate configurations
917  if (function != 0)
918  return false;
919 
920  // pin is already in default configuration, no change.
921  _pinData[pin].currentMux = DEFAULT_MUX;
922  return true;
923  }
924 }

928 //
929 // INPUTS:
930 // pin - The number of the Arduino GPI pin for which to select
931 // the multiplexed function.
932 // function - The multiplexer function to select for the pin:
933 // Either DEFAULT_MUX or ALTERNATE_MUX
934 // RETURN:
935 // true - The multiplexer is in the requested state.
936 // false - ALTERNATE_MUX was requested on a non-multiplexed pin.
937 //
938 // ERRORS:
939 // An error is thrown if any of the following occur:
940 // - A pin number is specified that is outside the valid range
941 // - An attempt is made to change the multiplexer state for a
942 // pin that is already locked for an incompatible use
943 // - An error is returned from a called routine.
944 //

946 {

950  {
951  _pinData[pin].muxSet = function;
952  return true;
953  }
954  return false;
955 }

1040  {

1042  // Cannot generate tones lower than 31Hz
1043  _ValidatePwmPin(pin);
1044  _ValidatePinOkToChange(pin);
1045  _InitializePinIfNeeded(pin);
1046

1048 
1049  // Scale the duty cycle to the range used by the driver.
1050  // From 0-255 to 0-PWM_MAX_DUTYCYCLE, rounding to nearest value.
1051  ULONG dutyCycle = ((255UL / 2 * PWM_MAX_DUTYCYCLE) + 127UL) / 255UL; // gives us a 50% duty cycle
1052 
1053  noTone(pin);
1054  // no tone running on pin
1055  // If PWM operation is not currently enabled on this pin:
1056  if (!_pinData[pin].pwmIsEnabled)

1061  _pinData[pin].stateIsKnown = FALSE;
